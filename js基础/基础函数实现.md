# 基础函数

## 防抖函数
    ```javascript
        function debunce = function(fn, delay) {
            let timer = null;
            return () => {
                timer && clearTimeOut(timer);
                timer = setTimeout(() => {
                    fn && fn();
                }, delay);
            }
            /**
             * return function() {
             *  let args = arguments;
             *  clearTimeout(timer);
             *  const context = this;
             *  timer = setTmeout(function() {
             *      fn && fn.apply(context, args)
             *  }, delay)
             * }
            */
        }

        var a = debunce(a, 300)();
    ```

## 截流函数 timeout 和时间戳两种实现方式
    ```javascript
        const th = function(fn, delay) {
            let timer = null;
            if (!timer) {
                setTimeout(function() {
                    fn();;
                    clearTimeout(timer);
                }, delay)
            }
        }
    ```
